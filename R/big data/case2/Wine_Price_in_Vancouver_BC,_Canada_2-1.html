数据整理

#新建因子变量what,存放红酒还是白酒
wine$what<-factor(wine$`Store Category Minor Name`)   
#新建因子变量where,存放酒的来源
wine$where<-factor(wine$`Store Category Sub Name`)    
#新建变量quantity,存放每周总销量
wine$quantity<-wine$`Total Weekly Selling Unit`
